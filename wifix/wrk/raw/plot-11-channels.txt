reset
filename = "sine-wave-reference.csv"
runtime = strftime("%Y-%m-%d %H:%M:%S",time(0))
unset multiplot
set terminal pdf size 7.5,10
set output "../post/plot-test.pdf"
set datafile separator ","
set key autotitle columnhead
do for [idx = 0:5] {
	set multiplot layout 2,1 title filename . " - " . runtime
	do for [pos = 0:1] {
		channel = 1 + (idx*2) + pos
		if (channel < 12) {
			set title sprintf("Channel %i",channel)
			plot filename using 0:channel with lines
		}
	}
	unset multiplot
}
reset

set output
set terminal pop
